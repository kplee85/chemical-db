How to connect Google Sheets

1

Open your Chemical_DB Google Sheet → Extensions → Apps Script

2

Delete any existing code. Paste the script below entirely:

const SHEET_NAME = "Chemical_DB"; const HEADERS = ["Timestamp","Brand Name","Active Ingredient","Formulation","Type","IRAC Group","FRAC Group","Mode of Action","PHI (days)","REI (hours)","Dosage Rate","Registered Crop","Manufacturer","Toxicity Class","Registration No.","Notes"]; function doPost(e) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME) || ss.insertSheet(SHEET_NAME); if (sheet.getLastRow() === 0) sheet.appendRow(HEADERS); const data = JSON.parse(e.postData.contents); const row = HEADERS.map(h => data[h] || ""); sheet.appendRow(row); return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService.createTextOutput(JSON.stringify({status:"error",message:err.toString()})).setMimeType(ContentService.MimeType.JSON); } } function doGet(e) { return ContentService.createTextOutput(JSON.stringify({status:"ok",message:"Chemical DB API running"})).setMimeType(ContentService.MimeType.JSON); }
3

Click Deploy → New deployment. Type: Web app. Execute as: Me. Access: Anyone. Click Deploy.

4

Copy the Web App URL and paste it in the field above. Click Test to verify.
